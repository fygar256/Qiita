たいちゃんは1000円で切手を買ってきなさいと言われました。６２円切手と８２円切手をどう買えばいいでしょう。


**************************************************


自然数、n1*m1+n2*m2の最適な組み合わせを求めます。
結果は最適順にソートして出力されます。

```kitte.cpp
#include    <iostream>
#include    <algorithm>
#include    <numeric>
#include    <vector>
#include    <string>
#include    <cstdlib>

using namespace std;

int main()
{
  int okane;
  int over;
  vector<int> data;
  int k;
  int k1;
  int k2;
  vector<string> str;
  char s[100];

  cout << "金額を入力して下さい：";
  cin >> okane;
  cout << "オーバーして良い金額を入力して下さい：";
  cin >> over;

  cout << "切手の金額を2個入力して下さい。\n";
  cout << "ひとつ目：";
  cin >> k1;
  cout << "ふたつ目：";
  cin >> k2;
  
  for (int i=0;i<=(okane+over)/k1;i++) {
    for(int j=0;j<=(okane+over)/k2;j++) {
      k=i*k1+j*k2;
      if (k<=okane+over) {
        sprintf(s,"合計%d円：%d円%d枚,%d円%d枚 差額%d円 %s",k,k1,i,k2,j,abs(okane-k),okane<k?"Over":"");
        str.push_back(s);
        data.push_back(abs(okane-k));
        }
      }
    }

    vector<size_t> indices(data.size());
    for(int i=0;i<data.size();i++)
      indices[i]=i;

    sort(indices.begin(), indices.end(), [&data](size_t i1, size_t i2) {
        return data[i1] < data[i2];
    });

    for(int i=0;i<indices.size();i++) {
      cout << str[indices[i]] <<endl;   // 表示
      }

  exit(0);
}
```
